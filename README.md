##Chú ý khi code
+ Vẽ hình bình hành
```
<?xml version="1.0" encoding="utf-8"?>
<selector xmlns:android="http://schemas.android.com/apk/res/android">
 <item>
    <rotate android:fromDegrees="116"
            android:pivotX="50%"
            android:pivotY="50%"
    >
        <shape android:shape="rectangle">
            <solid android:color="@color/colorBackground">
            </solid>
        </shape>
    </rotate>
 </item>
</selector>
```
